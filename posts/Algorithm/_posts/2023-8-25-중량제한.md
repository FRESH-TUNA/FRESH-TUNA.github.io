## 2. 풀이
```python
from collections import deque
import sys

def solution():
    input = sys.stdin.readline
    n, m = map(int,input().split())
    bridges = [[] for _ in range(n+1)]
    EDGES = [list(map(int,input().split())) for _ in range(m)]
    one, two = map(int,input().split())
    
    def bfs(weight): # weight == now
        queue = deque()
        queue.append(one)
        visited = [False] * (n+1)
        visited[one] = True

        while queue:
            x= queue.popleft() # w == limit

            for i,w in bridges[x]:
                if not visited[i] and w >= weight:
                    visited[i] = True
                    queue.append(i)

        return visited[two]

    def createBridge():
        for i in range(m):
            a,b,c = EDGES[i]
            bridges[a].append([b,c])
            bridges[b].append([a,c])

    def binarySearch():
        start, end = 1, 1000000000
        result = 0
        
        while start <= end:
            mid = (start + end) //2

            if bfs(mid):
                result = mid
                start = mid + 1
            else:
                end = mid - 1
        return result

    def main():
        createBridge()
        return binarySearch()

    return main()

print(solution())
```
